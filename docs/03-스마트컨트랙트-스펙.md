## 데이터 구조 (DTO)

```solidity
// VoucherProgram: 기관이 만드는 바우처 프로그램
struct VoucherProgram {
    uint16 programId;       // 프로그램 ID
    address issuer;         // 발급 기관 지갑 주소
    string name;            // 프로그램 이름
    uint256 amount;         // 바우처 금액 (원 단위)
    uint256 expiryDate;     // 만료 timestamp
    uint16 totalSupply;     // 총 발행 수량
    string category;        // 사용 가능 카테고리
}

// VoucherInfo: 개별 NFT 바우처
struct VoucherInfo {
    uint256 tokenId;        // NFT 토큰 ID
    uint16 programId;       // 어떤 프로그램인지
    string programName;     // 프로그램 이름
    uint256 amount;         // 잔액
    uint256 expiryDate;     // 만료일
    uint16 status;          // 1: 미사용, 2: 사용완료, 3: 만료
    address owner;          // 소유자
}

// MerchantInfo: 가맹점
struct MerchantInfo {
    address wallet;         // 가맹점 지갑 주소
    string name;            // 가맹점 이름
    string category;        // 카테고리 (음식점, 마트 등)
    bool isApproved;        // 승인 여부
}

// UseRecord: 사용 기록
struct UseRecord {
    uint256 tokenId;        // 어떤 바우처
    address merchant;       // 어떤 가맹점
    uint256 usedAmount;     // 사용 금액
    uint256 usedAt;         // 사용 시각 (timestamp)
}
```

## 주요 함수

### 기관이 호출하는 함수

```solidity
// 바우처 프로그램 생성
function createVoucherProgram(
    uint16 programId,
    string memory name,
    uint256 amount,
    uint256 expiryDate,
    uint16 totalSupply,
    string memory category
) public returns (uint16)

// 사용자에게 바우처 발급 (NFT 민팅)
function mintVoucher(
    uint16 programId,
    address recipient       // 받을 사람 지갑 주소
) public returns (uint256)  // 토큰 ID 반환

// 가맹점 등록 승인
function approveMerchant(address merchant) public
```

### 사용자가 호출하는 함수

```solidity
// 내 바우처 목록 조회 (사용 전)
function getMyActiveVouchers() public view returns (VoucherInfo[] memory)

// 내 바우처 목록 조회 (사용 완료)
function getMyUsedVouchers() public view returns (VoucherInfo[] memory)
```

### 가맹점이 호출하는 함수

```solidity
// 바우처 사용 처리
function useVoucher(
    uint256 tokenId,
    uint256 usedAmount      // 결제 금액
) public returns (bool)

// 바우처 유효성 검증 (QR 스캔 시)
function isValidVoucher(uint256 tokenId) public view returns (bool, VoucherInfo memory)
```

### 가맹점 등록

```solidity
// 가맹점 등록 신청
function registerMerchant(
    string memory name,
    string memory category
) public

// 가맹점 여부 확인
function isMerchant(address wallet) public view returns (bool)
```

## TTocket에서 바꾸는 것

| TTocket 함수 | 바우처 함수 | 변경 내용 |
|---|---|---|
| `createPerform()` | `createVoucherProgram()` | 좌석/시간 제거, 금액/카테고리 추가 |
| `createTicket()` | `mintVoucher()` | 결제 없음, 기관이 발급 |
| `cancleMyTicket()` | 제거 | 바우처는 취소 없음 |
| `buyCanceledTicket()` | 제거 | 재판매 없음 |
| `isOwnerOfPerform()` | `isValidVoucher()` | QR 검증용 |
| (없음) | `useVoucher()` | 신규: 가맹점 결제 처리 |
| (없음) | `registerMerchant()` | 신규: 가맹점 등록 |
